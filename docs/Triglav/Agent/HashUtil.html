def self.deep_stringify_keys(obj)
  case obj
  when Hash
    obj.map {|k, v| [k.to_s, deep_stringify_keys(v)] }.to_h
  when Array
    obj.map {|v| deep_stringify_keys(v) }
  else
    obj
  end
end

def self.deep_symbolize_keys(obj)

  when Array

end

def self.setdig(hash, key, val)
  keys = Array(key)
  sub_hash = hash
  keys[0...-1].each do |k|
    sub_hash = (sub_hash[k] ||= {})
  end
  sub_hash[keys.last] = val
  hash
end
